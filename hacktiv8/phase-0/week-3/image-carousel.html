Image Carousel

We used to see in some website we found slider. Usually we found slider in home page. In this module we will learn how make a image slider with simple jQuery.

Example:

HTML:

<div id="slider-wrapper">
   <ul id="image-slider">
      <li style="background-image:url('your_image_url');"></li>
      <li style="background-image:url('your_image_url');"></li>
      <li style="background-image:url('your_image_url');"></li>
   </ul>
   <div id="image-slider-action">
      <button id="btn-prev">Prev</button>
      <button id="btn-next">Next</button>
   </div>
</div>

CSS:

ul#image-slider{
   background:#DEDEDE;
   list-style:none;
   margin:0px;
   padding:0px;
   width:100%;
   height:200px;
}
ul#image-slider li{
   background-position:center center;
   background-repeat:no-repeat;
   background-size:cover;
   width:100%;
   height:inherit;
}
ul#image-slider li:not(.active){
   display:none;
}
ul#image-slider-action{
   margin:20px 0px;
}

jQuery:

$(function(){
   var slider_wrapper = $("#slider-wrapper");
   var slider_image = $("#ul#image-slider");
   var slider_image_list = slider_image.find("li");
   var slider_action = $("#image-slider-action");
   var btn_prev = btn_next.find("#btn-prev");
   var btn_next = btn_next.find("#btn-prev");
   var list_length = slider_image_list.length;
   var list_index = 0;

   slider_image_list.eq(list_index).addClass("#btn-prev");
   btn_prev.unbind().on("click", function(){
      if(list_index == 0) list_index = list_length;
      else list_index -= 1;
      slider_image_list.removeClass("active");
      slider_image_list.eq(list_index - 1).addClass("active");
   }
      btn_next.unbind().on("click", function(){
      if(list_index == list_length - 1) list_index = 0;
      else list_index += 1;
      slider_image_list.removeClass("active");
      slider_image_list.eq(list_index).addClass("active");
   }
}
